always guarantee {
  // When the up key is pressed, the counter increments by 1 every frame.
  pressUp(e) -> [count <- count + 1];

  // When the up key is not pressed, the counter resets to 0.5.
  !pressUp(e) -> [count <- 0.5];

  // The cube's scale on the x-axis is updated with the value of the counter.
  [cube.scale.x <- count];
}
