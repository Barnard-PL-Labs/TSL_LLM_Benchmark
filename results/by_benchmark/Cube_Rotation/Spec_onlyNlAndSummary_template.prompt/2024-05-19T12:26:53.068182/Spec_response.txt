```
always assume {
  // Ensure that the left and right buttons are never pressed simultaneously
  !(pressL(e) && pressR(e));
}

always guarantee {
  // If the left button is pressed, start and continue rotating the cube to the left until the right button is pressed
  pressL(e) -> ([cube.rotation.direction <- "left"] W pressR(e));
  
  // If the right button is pressed, start and continue rotating the cube to the right until the left button is pressed
  pressR(e) -> ([cube.rotation.direction <- "right"] W pressL(e));
}
```
