```tsl
always assume {
  // Mutual Exclusion for Key Inputs:
  !(pressL && pressR);
  !(pressL && pressUp);
  !(pressL && pressDown);
  !(pressR && pressUp);
  !(pressR && pressDown);
  !(pressUp && pressDown);
  // Ship Hit Detection:
  hit(ship) -> [dead(ship) <- 1];
}

always guarantee {
  // Movement and Actions:
  // Left Movement:
  (pressL && !leftmost(ship)) -> [ship <- moveLeft(ship)];
  // Right Movement:
  (pressR && !rightmost(ship)) -> [ship <- moveRight(ship)];
  // Shooting Action:
  pressS -> [ship <- shoot(ship)];
  // If ship is hit:
  hit(ship) -> [ship <- die(ship)];
}
``` 

This TSL specification details how the ship behaves based on player input and game events, ensuring that directional inputs are mutually exclusive and defining the ship's response to being hit.
