```tsl
always assume {
  // Ensure that no two directional keys can be pressed simultaneously
  !(pressL && pressR);
  !(pressL && pressUp);
  !(pressL && pressDown);
  !(pressR && pressUp);
  !(pressR && pressDown);
  !(pressUp && pressDown);
}

always guarantee {
  // If the ship is hit, signal the end game or lose life
  hit(ship) -> [die(ship)];

  // Move the ship left when the left key is pressed and not at the leftmost position
  pressL && !leftmost(ship) -> [moveLeft(ship)];

  // Move the ship right when the right key is pressed and not at the rightmost position
  pressR && !rightmost(ship) -> [moveRight(ship)];

  // Trigger shoot action when the shoot key is pressed
  pressS -> [shoot(ship)];
}
```
